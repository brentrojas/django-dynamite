    name = models.CharField(max_length = 255)
    slug = models.SlugField(unique = True)

    ENTITY_MODEL = NotImplemented
    ATTRIBUTE_MODEL = NotImplemented

    class Meta:
        abstract = True
        ordering = ('name',)

    def get_attributes_set(self):
        accessor_name = self.ATTRIBUTE_MODEL._meta.get_field('schema').related.get_accessor_name()
        return getattr(self, accessor_name).all()

    def __unicode__(self):
        return self.name

    @property
    def Entity(self):
        return self.get_entity(regenerate=False)

    @classmethod
    def bind(cls, entity_cls, attribute_cls):
        cls.ENTITY_MODEL = entity_cls
        cls.ATTRIBUTE_MODEL = attribute_cls

    def get_entity(self, regenerate=True):
        name = self.slug.encode('ascii', 'ignore')
        app_name = self.ENTITY_MODEL._meta.app_label
        model_name = '{0}_{1}'.format(self.ENTITY_MODEL.__name__, name)
        cached_model = cache.get_model(app_name, model_name)
        if cached_model is not None:
            if regenerate:
                 del cache.app_models[app_name][model_name.lower()]
            else:
                return cached_model
        attrs = {
            '__module__': getattr(self.ENTITY_MODEL, '__module__',  __name__),
        }
        class Meta:
            app_label = app_name
            verbose_name = self.name + ' ' + unicode(self.ENTITY_MODEL._meta.verbose_name)
            verbose_name_plural = self.name + ' ' + unicode(self.ENTITY_MODEL._meta.verbose_name_plural)
        attrs['Meta'] = Meta
        attributes = self.get_attributes_set()
        for attribute in attributes:
            field_name = attribute.slug.replace('-','_').encode('ascii', 'ignore')
            attrs[field_name] = attribute.get_field()
        attrs['_hash'] = self.generate_model_hash()
        attrs['_definition_model'] = self
        attrs['data'] = property(lambda s: tuple(getattr(s, q.slug) for q in attrs))
        model = type(model_name, (self.ENTITY_MODEL,), attrs)
        models.register_models(app_name, model)
        return model

    def generate_model_hash(self):
        val = simplejson.dumps([(a.slug, a.required, a.name, a.choices) for a in self.get_attributes_set()])
        return md5_constructor(val).hexdigest()


class AbstractAttribute(models.Model):
    schema = NotImplemented
    name = models.CharField(max_length=255, default="")
    slug = models.SlugField()
    attr_type = models.CharField(max_length=32, choices=fields.ATTR_TYPES)
    choices = models.CharField(max_length=1024, default="", blank=True,
        help_text="comma separated choices, keep them shortish")
    required = models.BooleanField(default=False)

    class Meta:
        abstract = True
        unique_together = ['schema', 'slug']

    def get_field(self):
        kwargs = {
            'blank' : not self.required,
            'verbose_name' : self.name,
        }
        if self.choices.strip():
            kwargs['choices'] = [(x.strip(), x.strip()) for x in self.choices.split(",")]
        try:
            return fields.ATTR_FIELDS[self.attr_type](**kwargs)
        except KeyError:
            return None

    @classmethod
    def factory(cls, schema):
        return type(
            'Attribute',
            (cls,),
            dict(
                schema = models.ForeignKey(schema),
                __module__ = getattr(schema, '__module__'),

            )
        )
